def fact(n):
    if n==1:
       return 1
    else:
       return n*fact(n-1)

def combination(n, k):
    if k==0:
        return 1
    else:
        if k==n:
            return 1
        else:
            return combination(n-1, k)+combination(n-1, k-1)
